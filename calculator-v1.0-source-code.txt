Public Sub Button16_Click()

  secondValue = Val(ValueBox1.text)
  
  If operatorId = 1 Then 
    ValueBox1.text = firstValue + secondValue
  Else If operatorId = 2 Then
    ValueBox1.text = firstValue - secondValue
  Else If operatorId = 3 Then
    ValueBox1.text = firstValue * secondValue
  Else If operatorId = 4 Then 
    ValueBox1.text = firstValue / secondValue
  Else 
    Message.Error("Error")  
  Endif

End
